-- Function to check if a user is whitelisted
local function is_user_whitelisted(username)
    for _, whitelisted_user in ipairs(whitelisted_users) do
        if username == whitelisted_user then
            return true  -- User is whitelisted
        end
    end
    return false  -- User is not whitelisted
end

-- Example usage

    io.write("Enter username to check: ")
